82 void SolucioNadal::actualitzarMaximesAversions(const int &convidat_actual, const posicio& pos_actual)
83 {
84  std::vector<int> veins = convidatsVeins(pos_actual);
85  if (!veins.empty())
86  {
87  if (_total_aversio != INT_MAX)
88  _total_aversio -= _max_aversions[convidat_actual];
